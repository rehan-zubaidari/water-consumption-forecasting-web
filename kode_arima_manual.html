Filter sesuai pilihan user logger.info(" Memfilter data...") df_filtered = df[df[filter_column] == filter_value].copy() df_filtered['TANGGAL'] = pd.to_datetime(df_filtered['TANGGAL']) df_filtered = df_filtered.sort_values(by='TANGGAL') df_filtered.set_index('TANGGAL', inplace=True) # Agregasi bulanan logger.info("Mengagregasi data bulanan...") df_filtered = df_filtered.groupby(pd.Grouper(freq='MS')).sum() df_filtered = df_filtered.ffill() series = df_filtered['KONSUMSI'] logger.info(f"Jumlah data setelah filter: {len(df_filtered)} baris") # Train-test split logger.info(" Membagi data train & test...") train_size = int(len(series) * 0.8) train, test = series[:train_size], series[train_size:] if len(series) < 20: raise ValueError("Data terlalu sedikit untuk ARIMA (min 20 bulan)") # Uji stasioneritas logger.info(" Menguji stasioneritas data...") def make_stationary(series): d = 0 while True: result = adfuller(series) if result[1] < 0.05: break series = series.diff().dropna() d += 1 if d > 3: break return series, d series_stationary, best_d = make_stationary(series) logger.info(f"Nilai d terpilih: {best_d}") # Decompose logger.info(" Membuat seasonal decomposition...") decomp = seasonal_decompose(series, model='additive') decomp.plot() plt.tight_layout() plt.savefig(f"static/decomposition_{filter_value}.png") plt.close() logger.info(f"Plot decomposition disimpan: static/decomposition_{filter_value}.png") # Grid search ARIMA logger.info(" Memulai grid search ARIMA...") best_mape = float("inf") best_aic = float("inf") best_cfg = None best_model = None for p, q in product(range(2, 5), range(2, 5)): try: model = ARIMA(train, order=(p, best_d, q)).fit() pred = model.predict(start=test.index[0], end=test.index[-1]) mape = mean_absolute_percentage_error(test, pred) * 100 aic = model.aic if (mape < best_mape) or (np.isclose(mape, best_mape) and aic < best_aic): best_mape = mape best_aic = aic best_cfg = (p, best_d, q) best_model = model except Exception as e: logger.debug(f"Gagal fit model ARIMA({p},{best_d},{q}): {e}") continue logger.info(f"Model baru terbaik ditemukan: {best_cfg} dengan MAPE {best_mape:.2f}% dan AIC {best_aic:.2f}") # Evaluasi model terbaik logger.info(" Mengevaluasi model terbaik...") pred = best_model.predict(start=test.index[0], end=test.index[-1]) evaluasi = { "Best_Order_(p,d,q)": best_cfg, "MAE": mean_absolute_error(test, pred), "MSE": mean_squared_error(test, pred), "RMSE": np.sqrt(mean_squared_error(test, pred)), "MAPE (%)": best_mape, "AIC": best_aic } logger.info(f"Hasil evaluasi: {evaluasi}") #
